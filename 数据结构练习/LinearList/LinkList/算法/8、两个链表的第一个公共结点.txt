ListNode* FindFirstCommonNode( ListNode* pHead1, ListNode* pHead2) {
		if(pHead1==nullptr||pHead2==nullptr)return nullptr;
        ListNode*h1=pHead1;
		ListNode*h2=pHead2;
		int CycleCount=0;
		while(h1!=h2)
		{
			if(h1->next)
			{
				h1=h1->next;
			}
			else {
				h1=pHead2;//换着跑另一个链表
				CycleCount++;
			}
			if(h2->next)
			{
				h2=h2->next;
			}
			else {
				h2=pHead1;
			}
			if(CycleCount>=2)return nullptr;//圈数>=2还没找到着表明无相较点
		}
		return h1;
    }
