bool IsBalanced_Solution(TreeNode* pRoot) {
        // write code here
        if(!pRoot)return true;
        bool IsBalanced=true;
        CalcHeight(pRoot, IsBalanced);
        return IsBalanced;
    }
    int CalcHeight(TreeNode*node,bool&bReturn)
    {
        if(node!=nullptr&&bReturn!=false)
        {
            int l=CalcHeight(node->left,bReturn);
            int r=CalcHeight(node->right,bReturn);
            if(abs(l-r)>1)bReturn=false;
            return max(l,r)+1;
        }
        return 0;
    }
