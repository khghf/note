vector<int> preorderTraversal(TreeNode* root) {
        // write code here
        vector<int>re;
        PreOrder(root,re);
        return re;
    }
    void PreOrder(TreeNode*node,vector<int>&arr)
    {
        if(node!=nullptr)
        {
            arr.push_back(node->val);
            PreOrder(node->left, arr);
            PreOrder(node->right, arr);
        }
        return;
    }
